When the parameters to the implementation of a partial method don't match those in the signature declaration, then confusion is almost guaranteed. Either the implementer was confused when he renamed, swapped or mangled the parameter names in the implementation, or callers will be confused.

Noncompliant Code Example

partial class Point
{
  partial void OnChange(int x, int y);
}

partial class Point
{
  int x = 0;
  int y = 0;

  partial void OnChange(int yk, int x)  // Noncompliant
  {
    this.x = x;
    this.y = y;
  }
}

Compliant Solution

partial class Point
{
  partial void OnChange(int x, int y);
}

partial class Point
{
  int x = 0;
  int y = 0;

  partial void OnChange(int x, int y)
  {
    this.x = x;
    this.y = y;
  }
}
